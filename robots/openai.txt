def askchatgpt(Question):
    response = openai.Completion.create(
    model="text-davinci-003",
    prompt=Q,
    temperature=0.7,
    max_tokens=256,
    top_p=1,
    frequency_penalty=0,
    presence_penalty=0
    )
    return response['choices'][0]['text'].lstrip()
